  {\Huge\filright}

% CAPITOLI
% --------------------------------
\titleformat{\chapter}[hang]
  {\Large\bfseries}
  {\thechapter.}
  {5pt}

\titlespacing*{\chapter}{0cm}{2.5cm}{1.5cm}

% SEZIONI
% --------------------------------
